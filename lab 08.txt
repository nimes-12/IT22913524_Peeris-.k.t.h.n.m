QUESTION 01 – Student Course Registration System


Course.java
public class Course {

    private String courseCode;
    private String courseTitle;

    // Constructor
    public Course(String courseCode, String courseTitle) {
        this.courseCode = courseCode;
        this.courseTitle = courseTitle;
    }

    // Display method
    public void displayCourseDetails() {

        System.out.println("Course Code : " + courseCode);
        System.out.println("Course Title : " + courseTitle);
    }
}
Student.java
public class Student {

    private String studentID;
    private String studentName;

    private Course[] courses = new Course[3];

    private int courseCount = 0;

    // Default constructor
    public Student() {

        studentID = "S000";
        studentName = "Unknown";
    }

    // Parameterized constructor
    public Student(String studentID, String studentName) {

        this.studentID = studentID;
        this.studentName = studentName;
    }

    // Enroll course
    public void enrollCourse(Course c) {

        if(courseCount < 3) {

            courses[courseCount] = c;
            courseCount++;
        }
        else {

            System.out.println("Cannot enroll more than 3 courses.");
        }
    }

    // Display student details
    public void displayStudentDetails() {

        System.out.println("===== Student Details =====");

        System.out.println("Student ID : " + studentID);
        System.out.println("Student Name : " + studentName);

        System.out.println("\nEnrolled Courses : ");

        for(int i = 0; i < courseCount; i++) {

            courses[i].displayCourseDetails();
            System.out.println();
        }
    }
}

StudentCourseApp.java

public class StudentCourseApp {

    public static void main(String[] args) {

        // Create courses
        Course c1 = new Course("SE101", "Java Programming");
        Course c2 = new Course("SE102", "Database Systems");
        Course c3 = new Course("SE103", "Web Development");

        // Create student
        Student s1 = new Student("IT241001", "Nimeshan");

        // Enroll courses
        s1.enrollCourse(c1);
        s1.enrollCourse(c2);
        s1.enrollCourse(c3);

        // Display details
        s1.displayStudentDetails();
    }
}



QUESTION 02 – Library Management System

Book.java

public class Book {

    private String isbn;
    private String title;

    private Library library;

    // Constructor
    public Book(String isbn, String title) {

        this.isbn = isbn;
        this.title = title;
    }

    // Setter method
    public void setLibrary(Library lib) {

        this.library = lib;
    }

    // Display details
    public void displayBookDetails() {

        System.out.println("ISBN : " + isbn);
        System.out.println("Title : " + title);

        if(library != null) {

            System.out.println("Library : " + library.getLibraryName());
        }
    }
}
Library.java
public class Library {

    private String libraryName;

    private Book[] books = new Book[3];

    private int bookCount = 0;

    // Constructor
    public Library(String libraryName) {

        this.libraryName = libraryName;
    }

    // Getter
    public String getLibraryName() {

        return libraryName;
    }

    // Add book
    public void addBook(Book book) {

        if(bookCount < books.length) {

            books[bookCount] = book;

            book.setLibrary(this);

            bookCount++;
        }
        else {

            System.out.println("Library is full.");
        }
    }

    // Display library details
    public void displayLibraryDetails() {

        System.out.println("===== Library Details =====");

        System.out.println("Library Name : " + libraryName);

        for(int i = 0; i < bookCount; i++) {

            books[i].displayBookDetails();

            System.out.println();
        }
    }
}
LibraryApp.java
public class LibraryApp {

    public static void main(String[] args) {

        // Create library
        Library lib = new Library("SLIIT Library");

        // Create books
        Book b1 = new Book("ISBN001", "Java Fundamentals");
        Book b2 = new Book("ISBN002", "Database Concepts");
        Book b3 = new Book("ISBN003", "Web Engineering");

        // Add books
        lib.addBook(b1);
        lib.addBook(b2);
        lib.addBook(b3);

        // Display details
        lib.displayLibraryDetails();
    }
}
QUESTION 03 – Printing System

Document.java


public class Document {

    private String title;
    private String content;

    // Constructor
    public Document(String title, String content) {

        this.title = title;
        this.content = content;
    }

    // Display document
    public void displayDocument() {

        System.out.println("Title : " + title);
        System.out.println("Content : " + content);
    }

    // Getter
    public String getTitle() {

        return title;
    }
}
Printer.java
public class Printer {

    public void printDocument(Document doc) {

        System.out.println("Printing document : " + doc.getTitle());

        doc.displayDocument();
    }
}
PrinterApp.java
public class PrinterApp {

        Document doc =
                new Document(
                        "Lab Report",

                );

    }
